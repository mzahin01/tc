class HomeView extends GetView<HomeController> {
  const HomeView({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Obx(() {
        return SingleChildScrollView(
          child: Padding(
            padding: const EdgeInsets.all(Spacing.xxlarge),
            child: CrossfadeWrapperContainer(
              visible: !controller.isLoading.value,
              loaderHeight: Get.height,
              //Pass respective controller to the widget
              child: ResponsiveWidget<HomeController>(
                pc: buildDesktopView(),
                tab: buildTabletView(),
                mobile: buildMobileView(),
              ),
            ),
          ),
        );
      }),
    );
  }

  
  // Mobile layout
  Widget buildMobileView() {
    return Center(
      child: const Text('Mobile View'),
    );
  }

  // Tablet layout
  Widget buildTabletView() {
    return Center(
      child: const Text('Tablet View'),
    );
  }

  // Desktop layout
  Widget buildDesktopView() {
    return Center(
      child: const Text('Desktop View'),
    );
  }
}
